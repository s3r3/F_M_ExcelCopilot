function copyToClipboard(text: string) {
  navigator.clipboard
    .writeText(text)
    .then(() => alert("âœ… Berhasil disalin ke clipboard!"))
    .catch((err) => console.error("âŒ Gagal menyalin:", err));
}

function csvToArray(csv: string): string[][] {
  const rows = csv.trim().split("\n");
  return rows.map((row) =>
    row
      .split(/,(?=(?:[^\"]*\"[^\"]*\")*[^\"]*$)/)
      .map((cell) => cell.trim().replace(/^['"]|['"]$/g, ""))
  );
}

function markdownToCsv(markdown: string): string {
  const lines = markdown
    .trim()
    .split("\n")
    .filter((line) => !/^(\s*\|[-\s|]+\|?\s*)$/.test(line));
  const csvLines = lines.map((line) =>
    line
      .trim()
      .replace(/^(\|)/, "")
      .replace(/(\|)$/, "")
      .split("|")
      .map((cell) => cell.trim())
      .join(",")
  );
  return csvLines.join("\n");
}

function pasteCsvToExcel(csv: string, givenRange?: Excel.Range) {
  const dataArray = csvToArray(csv);

  if (!dataArray.length || !Array.isArray(dataArray[0])) {
    alert("âŒ Format CSV tidak valid atau kosong.");
    return;
  }

  const cleanedData = cleanCurrency(dataArray); // â† bersihkan data sebelum ditempel

  Excel.run(async (context) => {
    const range = givenRange || context.workbook.getSelectedRange();
    const firstCell = range.getCell(0, 0);
    const numRows = cleanedData.length;
    const numCols = cleanedData[0].length;
    const targetRange = firstCell.getResizedRange(numRows - 1, numCols - 1);

    targetRange.values = cleanedData;

    const headerRange = firstCell.getResizedRange(0, numCols - 1);
    headerRange.format.fill.color = "#4472C4";
    headerRange.format.font.color = "white";
    headerRange.format.font.bold = true;

    for (let col = 0; col < numCols; col++) {
      const colRange = firstCell.getOffsetRange(0, col).getEntireColumn();
      colRange.format.autofitColumns();
    }

    const borderTypes = [
      Excel.BorderIndex.edgeTop,
      Excel.BorderIndex.edgeBottom,
      Excel.BorderIndex.edgeLeft,
      Excel.BorderIndex.edgeRight,
      Excel.BorderIndex.insideVertical,
      Excel.BorderIndex.insideHorizontal,
    ];

    borderTypes.forEach((borderType) => {
      const border = targetRange.format.borders.getItem(borderType);
      border.style = Excel.BorderLineStyle.continuous;
      border.weight = Excel.BorderWeight.thin;
      border.color = "black";
    });

    await context.sync();

    console.log("ðŸ“‹ CSV berhasil ditempel ke Excel dengan format:", cleanedData);
    alert(`âœ… Data berhasil ditempel (${numRows} baris Ã— ${numCols} kolom)`);
  }).catch((err) => {
    console.error("âŒ Gagal menempel ke Excel:", err);
    alert(
      "âŒ Gagal menempel ke Excel. Cek hal berikut:\nâ€¢ Range cukup luas\nâ€¢ Format CSV valid\nâ€¢ Sheet tidak terkunci."
    );
  });
}

function csvToTsv(csv: string): string {
  return csv
    .split("\n")
    .map((row) =>
      row
        .split(/,(?=(?:[^\"]*\"[^\"]*\")*[^\"]*$)/)
        .map((cell) => cell.replace(/^['"]|['"]$/g, "").replace(/"/g, '""'))
        .join("\t")
    )
    .join("\n");
}

async function createChartFromSelection(type: Excel.ChartType = Excel.ChartType.columnClustered) {
  await Excel.run(async (context) => {
    const sheet = context.workbook.worksheets.getActiveWorksheet();
    const range = context.workbook.getSelectedRange();
    const chart = sheet.charts.add(type, range, Excel.ChartSeriesBy.columns);
    chart.title.text = "ðŸ“Š Grafik Otomatis dari AI";
    chart.legend.position = Excel.ChartLegendPosition.right;
    chart.setPosition(range.getOffsetRange(2, 0), range.getOffsetRange(15, 6));
    chart.load("name");
    await context.sync();
    await showMessage(`âœ… Grafik '${type}' berhasil dibuat dan disisipkan!`, "success");
  }).catch(async (err) => {
    console.error("âŒ Gagal membuat grafik:", err);
    await showMessage("âŒ Gagal membuat grafik. Pastikan range valid dan memiliki data.", "error");
  });
}

function detectChartType(reply: string): Excel.ChartType | null {
  const lowerReply = reply.toLowerCase();

  // Ubah ke regex agar lebih fleksibel dan bisa deteksi meskipun AI menulis "grafik pie", "grafik garis", dll
  if (/bar|batang|kolom/.test(lowerReply)) return Excel.ChartType.columnClustered;
  if (/pie|lingkaran/.test(lowerReply)) return Excel.ChartType.pie;
  if (/line|garis/.test(lowerReply)) return Excel.ChartType.line;
  if (/scatter|sebar/.test(lowerReply)) return Excel.ChartType.xyscatter;
  if (/area/.test(lowerReply)) return Excel.ChartType.area;

  return null;
}

async function handleChartCreation(reply: string) {
  const detectedChartType = detectChartType(reply);
  if (detectedChartType) {
    console.log("ðŸŽ¯ Jenis grafik terdeteksi:", detectedChartType);
    await createChartFromSelection(detectedChartType);
  } else {
    console.log("âš ï¸ Tidak ditemukan jenis grafik dalam balasan AI.");
  }
}

  box.className = `status ${type}`; // misal styling CSS berdasarkan type
  box.style.display = "block";

}

function cleanCurrency(data: string[][]): string[][] {
  return data.map((row) =>
    row.map((cell) => {
      const cleaned = cell.replace(/Rp\.?\s?/gi, "").replace(/\./g, "");
      return /^[0-9]+$/.test(cleaned) ? cleaned : cell;
    })
  );
}
